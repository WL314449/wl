# JavaScript基础知识练习

```javascript
// 变量的定义
var name = "Alice"; 
let age = 30;
const PI = 3.14;

// 打印变量
console.log("Name:", name);
console.log("Age:", age); 
console.log("PI:", PI);

// 变量的类型
console.log("Name type:", typeof name);
console.log("Age type:", typeof age);
console.log("PI type:", typeof PI);

// 运算符与表达式（加上注释，请全面补全）
let a = 10;
let b = 5;

// 算数运算符
console.log("a + b =", a + b); // 加法
console.log("a - b =", a - b); // 减法
console.log("a * b =", a * b); // 乘法
console.log("a / b =", a / b); // 除法
console.log("a % b =", a % b); // 取模

// 赋值运算符
let c = 20; //赋值
c += a; // c = c + a
console.log("c after += a:", c);
c -= b; // c = c - b
console.log("c after -= b:", c);
c *= a; // c = c * a
console.log("c after *= a:", c);
c /= b; // c = c / b
console.log("c after /= b:", c);
c %= a; // c = c % a
console.log("c after %= a:", c);

//逻辑运算符
let x = true;
let y = false;
console.log("x && y =", x && y); // 与
console.log("x || y =", x || y); // 或
console.log("!x =", !x); // 非

//比较运算符
console.log("a > b =", a > b); // 大于
console.log("a < b =", a < b); // 小于
console.log("a >= b =", a >= b); // 大于等于
console.log("a <= b =", a <= b); // 小于等于
console.log("a == b =", a == b); // 等于
console.log("a != b =", a != b); // 不等于

//流程控制
if (a > b) {
    console.log("a is greater than b");
} else {
    console.log("a is not greater than b");
} 

for (let i = 0; i < 5; i++) {
    console.log("i:", i);
}

//循环控制
let j = 0;
while (j < 5) {
    console.log("j:", j);
    j++;
}  

//函数的定义与调用
function greet(name) {
    return "Hello, " + name + "!";
} 
console.log(greet("Alice"));

//javaScript中的数据类型
let str = "Hello, World!"; //字符串
let num = 42; //数字
let bool = true; //布尔值
let obj = { name: "Alice", age: 30 }; //对象
let arr = [1, 2, 3, 4, 5]; //数组
let undef; //undefined
let nul = null; //null

console.log("String:", str);
console.log("Number:", num);
console.log("Boolean:", bool);
console.log("Object:", obj);
console.log("Array:", arr);
console.log("Undefined:", undef);
console.log("Null:", nul);
```
